/**

 * CONTEXT:

 * 🎬 HEADED MODE: This test will run with visible browser
 * Run with: npx playwright test Verify_successful_search_for_an_English_article_and_redirection_.test.ts --headed
 */

import { test, expect } from '@playwright/test';
// Removed: import { LandingPage } from '../../page-objects/LandingPage'; // Not needed in Generic Mode
import { handlePageInterruptions } from '../../utils/pageInterruptions';


test.describe('Verify_successful_search_for_an_English_article_and_redirection_', () => {

  test.beforeEach(async () => {
      test.info().annotations.push({ type: 'Owner', description: 'AI Agent' });
      test.info().annotations.push({ type: 'Test Type', description: 'Generic' });
      test.info().annotations.push({ type: 'Description', description: 'Verify successful search for an English article and redirection.' });
  });

  test('Verify successful search for an English article and redirection.', { tag: ['@smoke', '@generic'] }, async ({ page }) => {
    // Removed: Initialize page object (LandingPage) as per Generic Mode guidelines
    // Removed: const inputData = {}; // Not used

    // =========================================
    // STEP 1: NAVIGATE TO PAGE: HTTPS://WWW.WIKIPEDIA.ORG/
    // =========================================
    console.log('\n🚀 STEP 1: Navigate to page: https://www.wikipedia.org/');

    console.log('📍 Navigating to URL: https://www.wikipedia.org/');
    await page.goto('https://www.wikipedia.org/', { waitUntil: 'domcontentloaded', timeout: 60000 });
    await expect(page.locator('body')).toBeVisible({ timeout: 15000 });
    await handlePageInterruptions(page); // Handle potential page interruptions (e.g., cookie banners)
    console.log('✅ Navigation completed successfully');
    console.log('✅ Step 1 completed');

    // =========================================
    // STEP 2: TYPE 'ARTIFICIAL INTELLIGENCE' INTO THE SEARCH INPUT FIELD WITH ID 'SEARCHINPUT'.
    // =========================================
    console.log('\n🚀 STEP 2: Type \'Artificial Intelligence\' into the search input field with id \'searchInput\'.');

    await handlePageInterruptions(page); // Handle potential page interruptions before interaction

    console.log('⌨️ Preparing to fill field: the search input field with id \'searchInput\'.');
    // Robust selector: Prioritizes semantic role and falls back to a stable ID.
    const searchInputField = page.getByRole('searchbox', { name: 'Search Wikipedia' }).or(page.locator('#searchInput'));
    await expect(searchInputField).toBeVisible({ timeout: 10000 });
    await searchInputField.fill('Artificial Intelligence');
    console.log('✅ Field filled: the search input field with id \'searchInput\'.');
    console.log('✅ Step 2 completed');

    // =========================================
    // STEP 3: ENSURE THE LANGUAGE DROPDOWN WITH ID 'SEARCHLANGUAGE' IS SET TO 'ENGLISH' (OR ITS DEFAULT VALUE).
    // =========================================
    console.log('\n🚀 STEP 3: Ensure the language dropdown with id \'searchLanguage\' is set to \'English\' (or its default value).');

    await handlePageInterruptions(page); // Handle potential page interruptions before interaction

    console.log('🔍 Verifying language dropdown setting.');
    // Robust selector for the language dropdown: Prioritizes semantic role and falls back to a stable ID.
    const languageDropdown = page.getByRole('combobox', { name: 'Search language' }).or(page.locator('#searchLanguage'));
    await expect(languageDropdown).toBeVisible({ timeout: 10000 });
    // Assert that the selected value of the dropdown is 'en' (which corresponds to English on Wikipedia).
    await expect(languageDropdown).toHaveValue('en');
    console.log('✅ Language dropdown verified to be set to \'English\'.');
    console.log('✅ Step 3 completed');

    // =========================================
    // STEP 4: CLICK THE 'SEARCH' BUTTON (ELEMENT WITH TYPE='SUBMIT').
    // =========================================
    console.log('\n🚀 STEP 4: Click the \'Search\' button (element with type=\'submit\').');

    await handlePageInterruptions(page); // Handle potential page interruptions before interaction

    console.log('🖱️ Preparing to click element: Search button.');
    // Robust selector for the search button: Prioritizes semantic role and falls back to common button types.
    const searchButton = page.getByRole('button', { name: /search/i }).or(page.locator('button[type="submit"]')).first();

    // Conditional click logic for robustness: if the button is not immediately visible, press Enter.
    if (await searchButton.isVisible({ timeout: 3000 }).catch(() => false)) {
      await searchButton.click();
    } else {
      console.log('⚠️ Search button not immediately visible, pressing Enter to submit the form.');
      await page.keyboard.press('Enter');
    }
    await page.waitForLoadState('domcontentloaded'); // Wait for the new page to load after search submission
    console.log('✅ Search submitted successfully');
    console.log('✅ Step 4 completed');

    // =========================================
    // VERIFICATIONS: POST-SEARCH REDIRECTION AND CONTENT
    // =========================================
    console.log('\n🔎 Performing post-search verifications.');

    await handlePageInterruptions(page); // Handle any potential interruptions on the newly loaded page

    // Verification 1: The browser navigates to 'https://en.wikipedia.org/wiki/Artificial_intelligence' (or a similar direct article page).
    console.log('✅ Verifying URL redirection to the article page.');
    await expect(page).toHaveURL(/en\.wikipedia\.org\/wiki\/Artificial_intelligence/i, { timeout: 15000 });

    // Verification 2: The page title contains 'Artificial intelligence - Wikipedia'.
    console.log('✅ Verifying page title contains expected text.');
    await expect(page).toHaveTitle(/Artificial intelligence - Wikipedia/i, { timeout: 10000 });

    // Verification 3: The main heading of the article is 'Artificial intelligence'.
    console.log('✅ Verifying main article heading is correct.');
    // Use getByRole for semantic heading check (h1).
    const mainArticleHeading = page.getByRole('heading', { level: 1, name: 'Artificial intelligence' });
    await expect(mainArticleHeading).toBeVisible({ timeout: 10000 });
    await expect(mainArticleHeading).toHaveText('Artificial intelligence');

    console.log('✅ All verifications completed successfully.');
  });
});

export {}; // Make this a module
